/* Canadian credit card rewards value analysis */
WITH card_base AS (
    SELECT
        card_id,
        card_name,
        issuer,
        annual_fee,
        income_requirement,
        CASE WHEN province = 'QC' THEN 
            welcome_bonus_non_quebec 
        ELSE 
            welcome_bonus 
        END AS welcome_bonus,
        CASE WHEN foreign_transaction_fee = 0 THEN 1 ELSE 0 END AS no_fx_fee
    FROM canadian_credit_cards
    WHERE is_active = TRUE
    AND (income_requirement <= :annual_income OR income_requirement IS NULL)
    AND (:no_fee_only = 0 OR annual_fee = 0)
),
reward_rates AS (
    SELECT
        card_id,
        MAX(CASE WHEN category = 'Groceries' THEN reward_rate ELSE 0 END) AS grocery_rate,
        MAX(CASE WHEN category = 'Dining' THEN reward_rate ELSE 0 END) AS dining_rate,
        MAX(CASE WHEN category = 'Gas' THEN reward_rate ELSE 0 END) AS gas_rate,
        MAX(CASE WHEN category = 'Travel' THEN reward_rate ELSE 0 END) AS travel_rate,
        MAX(CASE WHEN category = 'Recurring' THEN reward_rate ELSE 0 END) AS recurring_rate,
        MAX(CASE WHEN category = 'Other' THEN reward_rate ELSE 0 END) AS other_rate,
        reward_type,
        point_value
    FROM card_reward_rates
    GROUP BY card_id, reward_type, point_value
),
user_spending AS (
    SELECT
        :groceries AS groceries,
        :dining AS dining,
        :gas AS gas,
        :travel AS travel,
        :recurring AS recurring,
        :other AS other,
        :groceries + :dining + :gas + :travel + :recurring + :other AS monthly_total,
        (:groceries + :dining + :gas + :travel + :recurring + :other) * 12 AS annual_total
),
reward_value AS (
    SELECT
        cb.card_id,
        cb.card_name,
        cb.issuer,
        cb.annual_fee,
        CASE WHEN :include_welcome_bonus = 1 THEN cb.welcome_bonus ELSE 0 END AS welcome_bonus,
        cb.no_fx_fee,
        rr.reward_type,
        rr.point_value,
        us.monthly_total,
        us.annual_total,
        (us.groceries * 12 * rr.grocery_rate * rr.point_value) +
        (us.dining * 12 * rr.dining_rate * rr.point_value) +
        (us.gas * 12 * rr.gas_rate * rr.point_value) +
        (us.travel * 12 * rr.travel_rate * rr.point_value) +
        (us.recurring * 12 * rr.recurring_rate * rr.point_value) +
        (us.other * 12 * rr.other_rate * rr.point_value) AS annual_rewards,
        CASE WHEN :travel_value = 1 AND rr.reward_type = 'Travel' THEN
            (us.travel * 12 * 0.015) /* Extra 1.5% value for travel-specific cards when user values travel */
        ELSE 0 END AS travel_premium,
        CASE WHEN cb.no_fx_fee = 1 THEN
            (us.travel * 12 * 0.025) /* 2.5% savings on foreign transactions for no-FX-fee cards */
        ELSE 0 END AS fx_fee_savings
    FROM card_base cb
    JOIN reward_rates rr ON cb.card_id = rr.card_id
    CROSS JOIN user_spending us
)

SELECT
    card_name,
    issuer,
    annual_fee,
    reward_type,
    annual_rewards,
    welcome_bonus_value,
    travel_premium,
    fx_fee_savings,
    annual_rewards + welcome_bonus_value + travel_premium + fx_fee_savings AS total_value,
    annual_rewards + welcome_bonus_value + travel_premium + fx_fee_savings - annual_fee AS net_value,
    ROUND((annual_rewards / annual_total) * 100, 2) AS effective_rate,
    RANK() OVER (ORDER BY (annual_rewards + welcome_bonus_value + travel_premium + fx_fee_savings - annual_fee) DESC) AS card_rank
FROM reward_value
ORDER BY net_value DESC
LIMIT 10;
